                        # 简介

                        • EditTable 仅配置columns就可以完成一个可编辑的表格
                        • ProTable 仅配置columns就可以展现一个带搜索、带分页的表格

                        # 背景

                        • 减少重复代码逻辑，将工作集中于业务逻辑上
                        • 开发快捷高效，将时间用于业务代码上
                        • 经常用到删除、添加、单元格可编辑的表格业务

                        # 特性

                        • 灵活度高: 使用jsx语法编写，除通用基础表单类型，给开发者提供了更灵活的自定义的写法，所有代码逻辑均可配置

                        # 使用

                        • js中引入组件 import EditTable from '@/component/XTable/EditTable
                        • html中注入组件 <EditTable/>

                        # 共有的属性

                        属性名默认值类型描述用法
                        form{}Object双向绑定v-model:form绑定的值比如你再js中已经有了const state=reactive({form:{}});
                        那么在组件属性中添加v-model:form={state.form}实现表格值双向绑定
                        columns[]Array表格的表头以及对应列表单类型字段配置的数组具体参见columns属性columns="[{label:'姓名',prop:'name'},...]"
                        rowSelectionnullObject是否可选择行,值为对应的配置项
                        其他选择相关属性均可添加                        		rowSelection="{rowKey: "id",onSelectionChange:val=>{}}"
                        toolBarRendernullObject表格上方固定的操作按钮，自动右对齐表格，多个间会有间距，接收一个元素数组
                        		toolBarRender={[<el-button>下载模版</el-button>,<el-button>导入用户</el-button>]}
                        tableAlertRendernullObject表格上方固定提示内容,会自动有背景及其间距样式（如：表格内容不分页的合计）tableAlertRender={<p>总销售金额：100</p>}
                        summarynullObject表格最后一行的汇总
                        showSummary为true时展示汇总行，summaryMethod不传时，默认会将columns配置项中summary属性为逻辑true的那列进行合计                        		summary="{showSummary: true,summaryMethod:()=>[]
                        expandnullDOM Element嵌套表格的插槽将会接收两个参数row 当前行数据项, index 当前列号，一个表格组件，expand="{default:()=><EditTable/>"
                        pagination看描述Object默认值{pageNum:1,pageSize:7,total:1}，
                        分页配置，当pagination={false}时不展示分页                        		pagination="{pageNum:1,pageSize:15,total:1}"
                        requestnullFunction动态获取数据的异步方法，会在组件初始化、切换pagination分页时、搜索search查询时自动触发callback方法，并将参数带回,具体参见request属性request={params=>{}}
                        paramsnullObjectrequest请求的额外参数，会在request触发时，跟分页和查询的参数一起返回到callback方法入参params="{isBigContract:true}"
                        manualRequesttrueBoolean是否手动请求接口，为true则不会初始化时触发request请求manualRequest={true}
                        searchtrueBoolean是否显示表格上方的搜索表单，默认会将columns配置项中hideInSearch属性值不为true、valueType属性值不为index、action的其他所有列作为搜索项显示在搜索表单，并以valueType的类型(默认为input)为表单类型search={false}
                        labelWidth130Number｜String搜索表单的labelWidth属性labelWidth={200}
                        flex4Number搜索表单的列数，默认4，可传值遵循element ui 栅栏
                        searchConfignullArray自定义的search配置searchConfig="[{label:'姓名',prop:'name',valueType:'select'}]"
                        headerTitle""String表格标题headerTitle="用户列表"
                        exportConfignullObject表头显示报表导出按钮exportConfig="{url:'接口地址',id:'权限id'}"
                        optionsfalseBoolean｜Array表格配置项默认为false不展示,可以设置为true展示，也可以传需要展示的配置项数组，如：['setting,'reload']
                        # request属性
                        <EditTabel
    params={{type: query.type}}
    request={async (params) => {
        const res = await getUserList({...params, code: '001'})
        const {pageInfo} = res
        return {
            data: pageInfo.list, //必须return一个data出去，将会自动作为tabel的data属性值
            pagination: {        //当pagination不为false时，必须return一个data出去，将会自动计算总页数，和当前页数
                pageNum: pageInfo?.pageNum,
                pageSize: pageInfo?.pageSize,
                total: pageInfo?.total
            }
        }
    }}/>

                        # 共有的属性

                        属性名类型描述用法
                        labelString列标题，同时当表格的search属性不为false，且当前项hideInSearch属性值不为true时，也将作为搜索表单的表单label字段值label:"姓名"
                        propString列字段名称，同时当表格的search属性不为false，且当前项hideInSearch属性值不为true时，也将作为搜索表单的表单prop字段值label:"name"
                        hideInSearchBoolean该项是否为搜索项，默认false，当为true时该项将不会出项在搜索表单hideInSearch:true
                        hideInTableBoolean该项是否在表单中隐藏，默认false，当为true时该项将不会出项在表格中hideInTable:true
                        fieldString当表格显示所用prop与搜索表单所用prop不一致时，表单将优先使用field作为搜索表单的表单prop字段值field:"userName"
                        valueTypeString搜索表单的类型，默认为input，具体参看 valueType属性valueType:"select"
                        sortBoolean｜Function该列是否可排序，默认为false，也可以自己传递排序方法 比如：调用接口
                        searchRendernullFunction自定义的搜索表单，jsx语法 (scope)=>{}
                        renderFunction自定义列单元格显示内容，该方法将得到一个callback方法，3个参数,具体参见 render属性render:(item,row,index)=>{}
                        slot-Object表格每列插槽
                        summaryObject｜Boolean当为逻辑true时该列将汇总所有值，也可接收一个配置对象,具体参见 summary属性summary:true

                        属性名描述
                        select自定义选择
                        asyncSelect异步选择 ，具体参见对应插件
                        enumsSelect静态枚举 ，具体参见对应插件

                        date日期
                        time时间

                        upload文件
                        input输入框 ，默认值，为此时可不写
                        number数值输入框
                        textarea文本输入
                        action操作
                        switch切换按钮
                        numberString数字字符，可以0开头的数值

                        # EditTable独有的columns属性

                        • EditTable的columns支持其valueType 属性值对应的element plus Formopen in new window中表单类型对应的所有的属性
                        • 以下所列是基于其原有属性的扩展
                        属性名类型描述用法
                        onChangeFunction单元格的change事件，3个参数,具体参见onChange属性onChange:(val,row,index) => {}
                        editBoolean｜Function当前单元格是否是可编辑的表单，默认为true，为false时将直接渲染成对应值，同样3个参数edit:(item,row,index)=>row.status===1
                        disabledBoolean｜Function当前单元格表单disabled值，同样3个参数disabled:(item,row,index)=>row.status===1
                        maxNumber｜FunctionvalueType:'number'最大值，同样3个参数max:(item,row,index)=>row.maxNum
                        minNumber｜FunctionvalueType:'number'最小值，同样3个参数min:(item,row,index)=>row.minNum
                        # summary属性
                        {
    summary:{
        sumText:"总回款金额",   // sumText：汇总合计文案，默认为'汇总
        prop:"amount",        // prop：参与汇总计算字段名，默认为当前项的'prop'值 
        precision:2           // precision：汇总计算总值保留小数位数，默认0位 
    }
}
                        # render属性
                        {
    render:(item, row, index) => {
        return (
            <el-button type="primary"
                       onClick={() => func({id: row.id})}
            >激活</el-button>
        )
    }
}
// item：当前配置项对象  
// row：当前行数据项

                        属性名类型描述用法
                        nameString操作项名称name:"新增"

                        # EditTable & ProTable方法

                        # 共有的方法

                        方法名接收参数类型描述用法
                        reloadAndRestObject查询参数 、分页参数重置并重新执行requestrefs.value?.reload({type:1})
                        reloadObject执行request(查询参数 、分页参数不会重置)refs.value?.reloadAndRest({type:1})
                        summaryMethodObject(即summary-method方法的参数)获取汇总得到的结果
                        （{summaryArr:,summaryObj:}）                        		refs.value?.summaryMethod()

                        具体用法

                        const refs = ref(null)

//1.加载：当需要查询条件和分页保持不变的刷新时可以用这个方法
const reload = () => {
    refs.value?.reload({type: 1})
}

//2.刷新：当需要重置查询条件和分页并加载时可以用这个方法
const reloadAndRest = () => {
    refs.value?.reloadAndRest({type: 1})
}

//3.计算剩余吨数：当你需要使用计算结果去做其他计算时可以用这个方法
const surplus = computed(() => {
    /* summaryMethod方法返回一个对象，两个属性
    1.summaryArr是需要展示在表格汇总行的数组，形如：['汇总',12.00,23.0989]  
    （这个会在表格属性summary={showSummary:true}时自动展示，项目中一般用不上）
    2.summaryObj是所有需要计算的列字段集合对象，形如：{mergeProp1:12.00,mergeProp2:23.0989}
    */
    const res = refs.value?.summaryMethod()?.summaryObj
    return summaryObj.weightA - summaryObj.weightB
})

return () => <div>
    <ProTable ref={refs}/>
    <button onClick={reload}>1.加载</button>
    <button onClick={reloadAndRest}>2.刷新</button>
    <div>3.当前剩余吨数为：{surplus.value}</div>
</div>
